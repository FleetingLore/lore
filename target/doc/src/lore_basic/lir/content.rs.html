1//! LIR 内容解析模块
2//!
3//! 这个模块负责解析 LIR (Lore Intermediate Representation) 中的单行内容，
4//! 将原始文本区分为两种类型：普通元素和域定义。
5//!
6//! ## 核心概念
7//!
8//! LIR 使用简单的语法规则来区分内容类型：
9//!
10//! - **普通元素 (Element)**: 普通的文本行，不包含特殊前缀
11//! - **域定义 (Domain)**: 以 "+ " 开头的行，表示一个可以包含子元素的域
12//!
13//! ## 解析规则
14//!
15//! 1. 如果一行以 "+ " 开头（加号后必须有空格），则解析为 `Content::Domain`
16//! 2. 所有其他情况都解析为 `Content::Element`
17//! 3. 空字符串解析为空的 `Content::Element`
18//!
19//! ## 示例
20//!
21//! ```text
22//! 普通文本          -> Element("普通文本")
23//! + 域定义          -> Domain("域定义")
24//! +no space        -> Element("+no space")  // 注意：缺少空格
25//! +                -> Element("+")          // 注意：缺少空格和内容
26//! +                -> Domain("")            // 注意：有空格但无内容
27//! ```
28//!
29//! ## 主要组件
30//!
31//! - [`Content`] 枚举: 表示两种内容类型
32//! - [`parse_content`] 函数: 从字符串解析内容类型
33//! - `Display` 实现: 用于格式化输出
34//!
35//! 这个模块与 [`crate::lir::line`] 和 [`crate::lir::root`] 模块协同工作，共同完成 LIR 文档的完整解析。
36
37/// 表示 LIR 中的单行内容类型。
38///
39/// 根据 LIR 语法规范，内容被划分为两种基本类型：
40/// - 普通元素：包含任意文本内容
41/// - 域定义：以 "+ " 前缀标识，可以包含子元素
42///
43/// # 生命周期
44///
45/// 使用生命周期 `'f` 来借用原始字符串切片，避免不必要的内存分配。
46///
47/// # 示例
48///
49/// ```
50/// use crate::lir::content::{Content, parse_content};
51///
52/// let element = parse_content("普通文本");
53/// assert!(matches!(element, Content::Element("普通文本")));
54///
55/// let domain = parse_content("+ 域定义");
56/// assert!(matches!(domain, Content::Domain("域定义")));
57/// ```
58pub enum Content<'f> {
59    /// 普通元素，包含任意文本内容
60    ///
61    /// 普通元素是 LIR 中的基本构建块，可以出现在任何缩进级别。
62    /// 它们不包含特殊的语法含义，只是纯文本内容。
63    Element(&'f str),
64
65    /// 域定义，以 "+ " 前缀标识
66    ///
67    /// 域是一种特殊的内容类型，它可以包含子元素（普通元素或其他域）。
68    /// 域为 LIR 文档提供了层次结构的能力。
69    Domain(&'f str)
70}
71
72/// 为了便于测试，这里给 `Content<'f>` 实现 `Display`。
73impl<'f> std::fmt::Display for Content<'f> {
74    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
75        match &self {
76            Content::Element(tokens) => write!(f, "{}", tokens),
77            Content::Domain(tokens) => write!(f, "+ {}", tokens)
78        }
79    }
80}

82/// 解析字符串内容并返回相应的 `Content` 类型。
83///
84/// 这个函数是内容解析的核心，它根据简单的前缀规则判断内容类型。
85///
86/// # 参数
87///
88/// - `content`: 要解析的字符串切片，应该是不包含前导空格的纯内容
89///
90/// # 返回值
91///
92/// 返回 `Content` 枚举，表示解析出的内容类型。
93///
94/// # 规则
95///
96/// 1. 如果内容以 "+ " 开头 → `Content::Domain(剩余内容)`
97/// 2. 其他所有情况 → `Content::Element(原内容)`
98///
99/// # 注意
100///
101/// 这个函数期望接收已经去除前导空格的内容。在实际使用中，
102/// 通常先由 [`crate::lir::line::parse_line`] 处理缩进，然后将剩余内容传递给此函数。
103///
104/// # 示例
105///
106/// ```
107/// use crate::lir::content::{parse_content, Content};
108///
109/// assert!(matches!(parse_content("hello"), Content::Element("hello")));
110/// assert!(matches!(parse_content("+ world"), Content::Domain("world")));
111/// assert!(matches!(parse_content("+no space"), Content::Element("+no space")));
112/// ```
113pub fn parse_content(content: &str) -> Content<'_> {
114    if let Some(rest) = content.strip_prefix("+ ") {
115        Content::Domain(rest)
116    } else {
117        Content::Element(content)
118    }
119}

121/// 测试模块如此。测试结果与预期一致。

126    // 现有测试保持不变...
127    #[test]
128    fn test_element_content() {
129        let content = parse_content("hello world");
130        match content {
131            Content::Element(text) => assert_eq!(text, "hello world"),
132            _ => panic!("Expected Element variant"),
133        }
134    }
135
136    #[test]
137    fn test_domain_content() {
138        let content = parse_content("+ domain content");
139        match content {
140            Content::Domain(text) => assert_eq!(text, "domain content"),
141            _ => panic!("Expected Domain variant"),
142        }
143    }
144
145    #[test]
146    fn test_domain_content_but_no_space_after_plus() {
147        let content = parse_content("+no space");
148        match content {
149            Content::Element(text) => assert_eq!(text, "+no space"),
150            _ => panic!("Expected Element variant"),
151        }
152    }
153
154    #[test]
155    fn test_empty_string() {
156        let content = parse_content("");
157        match content {
158            Content::Element(text) => assert_eq!(text, ""),
159            _ => panic!("Expected Element variant"),
160        }
161    }
162
163    #[test]
164    fn test_only_plus_and_a_space() {
165        let content = parse_content("+ ");
166        match content {
167            Content::Domain(text) => assert_eq!(text, ""),
168            _ => panic!("Expected Domain variant"),
169        }
170    }
171
172    #[test]
173    fn test_display_implementation() {
174        let element = Content::Element("test element");
175        let domain = Content::Domain("test domain");
176
177        assert_eq!(format!("{}", element), "test element");
178        assert_eq!(format!("{}", domain), "+ test domain");
179    }

181    #[test]

183        let test_cases = vec![
184            ("normal text", "normal text"),
