1use std::fs;
2use std::path::Path;
3
4// Lore IR 是被解析的 Lore 代码的 rust 数据对象
5pub mod lir;
6use lir::root::*;
7
8/// 文件类，作为原始数据的所有者
9pub struct File {
10    path: String,
11    content: Vec<String>,
12}
13
14impl File {
15    // 获取文件数据
16    pub fn new(file_path: &str) -> Self {
17        // 文件不存在
18        if !Path::new(file_path).exists() {
19            panic!("文件不存在: {}", file_path);
20        }
21
22        // 读取文件
23        let content_str = fs::read_to_string(file_path)
24            .expect("读取文件失败");
25        let content: Vec<String> = content_str
26            .as_str()
27            .lines()
28            .map(|line| line.to_string())
29            .collect();
30
31        // 返回文件数据
32        File {
33            path: file_path.to_string(),
34            content,
35        }
36    }
37
38    // 数据解析
39    pub fn parse<'f>(self) -> Root<'f> {
40        // 解析每一行
41        let lines: Vec<&'f str> = self
42            .content
43            .into_iter()
44            .map(|s| {
45                let boxed_str = s.into_boxed_str();
46                let leaked: &'f mut str = Box::leak(boxed_str);
47                &*leaked
48            })
49            .collect();
50
51        // 生成 root
52        parse_root(lines)
53    }
54}
55
56// 给文件名，返回解析好的数据
57pub fn read_and_parse_file<'f>(file_path: &str) -> Root<'f> {
58    File::new(file_path).parse()
59}
60
61// 便于检查文件的基本信息
62impl std::fmt::Display for File {
63    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
64        write!(f, "File: {} ({} lines)", self.path, self.content.len())
65    }
66}
